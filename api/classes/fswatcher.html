表示一个文件系统监听器。

example

var watcher = new FSWatcher(); watcher.on("change", function (path){ console.log('Changed', path) }); watcher.on("delete", function (path){ console.log('Deleted', path) }); watcher.on("create", function (path){ console.log('Created', path) }); watcher.add(".", function() { console.log('Start Watching...') });
